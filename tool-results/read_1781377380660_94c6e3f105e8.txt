    44→/**
    45→ * Account Security Screen — Passkey Enrollment Engine
    46→ *
    47→ * Implements the full passkey creation workflow with defensive domain validation:
    48→ *
    49→ * ARCHITECTURE FLOW:
    50→ * 1. Fetch WebAuthn challenge from AuthRepository (Cloudflare Edge Worker or local fallback)
    51→ * 2. DEFENSIVE: Validate and enforce rp.id = "zixo.pages.dev" in the JSON payload
    52→ * 3. Create CreatePublicKeyCredentialRequest with preferImmediatelyAvailableCredentials = false
    53→ * 4. Invoke CredentialManager.createCredential() on IO thread
    54→ * 5. Parse CreatePublicKeyCredentialResponse — extract credential ID
    55→ * 6. Register credential with backend + save to Firestore
    56→ *
    57→ * ERROR HANDLING:
    58→ * - CreateCredentialCancellationException → "Registration cancelled or domain mismatch"
    59→ * - CreatePublicKeyCredentialDomException → DOM-level WebAuthn validation error with detailed code
    60→ * - CreateCredentialCustomException → assetlinks.json configuration issue
    61→ * - All others → Generic WebAuthn error with diagnostic message
    62→ */

   202→                                                    ?: "Failed to get challenge from server"

   245→                                                }

   274→                                            }

   278→                                                passkeyError = "Custom validation hook failure. Verify that https://zixo.pages.dev/.well-known/assetlinks.json is accessible and correctly configured."
   279→                                            }
   280→                                        } catch (e: CreateCredentialInterruptedException) {
   281→                                            Log.e("Passkey", "Credential creation interrupted", e)
   282→                                            withContext(Dispatchers.Main) {
   283→                                                passkeyError = "Passkey creation was interrupted. Please try again."
   284→                                            }
   285→                                        } catch (e: CreateCredentialProviderConfigurationException) {
   286→                                            Log.e("Passkey", "Provider not configured", e)
   287→                                            withContext(Dispatchers.Main) {
   288→                                                passkeyError = "Credential provider not available. Ensure Google Play Services is updated."
   289→                                            }
   290→                                        } catch (e: CreateCredentialException) {

   298→                                                        "(2) SHA-256 fingerprint matches your signing certificate, " +
   299→                                                        "(3) Digital Asset Links verification is enabled in Firebase."
   300→                                                } else {

   303→                                            }

   581→/**
   582→ * CLIENT-SIDE DEFENSIVE RP ID ENFORCEMENT
   583→ *
   584→ * This function acts as a final safety net before the challenge JSON
   585→ * is passed to Android's CredentialManager. It ensures that:
   586→ *
   587→ * 1. The "rp.id" field is exactly "zixo.pages.dev"
   588→ * 2. The "rp.name" field is "Zixo"
   589→ * 3. The "user.id" field is properly Base64Url encoded
   590→ * 4. The "challenge" field is properly Base64Url encoded
   591→ *
   592→ * This prevents the "rp id cannot be validated" error that occurs
   593→ * when a backend returns an incorrect or malformed rp.id value.
   594→ */
   595→private fun enforceRpIdOnClient(challengeJson: String): String {
   596→    return try {
   597→        val json = JSONObject(challengeJson)
   598→
   599→        // ── Force rp.id compliance ──
   600→        val rp = json.optJSONObject("rp") ?: JSONObject()
   601→        rp.put("id", AuthRepositoryImpl.RP_ID)
   602→        rp.put("name", AuthRepositoryImpl.RP_NAME)
   603→        json.put("rp", rp)
   604→
   605→        // ── Ensure user.id is Base64Url encoded ──
   606→        val user = json.optJSONObject("user")
   607→        if (user != null) {
   608→            val userId = user.optString("id", "")
   609→            if (userId.isNotEmpty()) {
   610→                // Check if it looks like a raw UUID/string (contains hyphens or isn't Base64Url)
   611→                val isRawString = userId.contains("-") || !userId.all { it.isLetterOrDigit() || it == '_' || it == '-' }
   612→                if (isRawString) {
   613→                    val encoded = android.util.Base64.encodeToString(
   614→                        userId.toByteArray(Charsets.UTF_8),
   615→                        android.util.Base64.URL_SAFE or android.util.Base64.NO_PADDING or android.util.Base64.NO_WRAP
   616→                    )
   617→                    user.put("id", encoded)
   618→                }
   619→            }
   620→        }
   621→
   622→        // ── Ensure challenge is Base64Url encoded ──
   623→        val challenge = json.optString("challenge", "")
   624→        if (challenge.isNotEmpty()) {
   625→            val isNotBase64Url = challenge.contains("+") || challenge.contains("/") || challenge.contains("=")
   626→            if (isNotBase64Url) {
   627→                val encoded = challenge
   628→                    .replace("+", "-")
   629→                    .replace("/", "_")
   630→                    .replace("=+$".toRegex(), "")
   631→                json.put("challenge", encoded)
   632→            }
   633→        }
   634→
   635→        json.toString()
   636→    } catch (e: Exception) {
   637→        Log.e("Passkey", "Client-side RP ID enforcement failed, using original JSON", e)
   638→        challengeJson
   639→    }
   640→}
