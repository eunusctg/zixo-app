    50→    override suspend fun getOrCreateChat(otherUserId: String): Result<String> {
    51→        return try {
    52→            val uid = auth.currentUser?.uid ?: return Result.failure(Exception("Not authenticated"))
    53→            if (uid == otherUserId) return Result.failure(Exception("Cannot chat with yourself"))
    54→
    55→            // ── Zero-Trust Contact Gate ──
    56→            if (contactRepository != null) {
    57→                val verification = contactRepository.verifyContactGate(otherUserId)
    58→                if (verification !is ContactVerification.MutualContact) {
    59→                    return Result.failure(Exception("Cannot message — not a mutual contact. Add this user first."))
    60→                }
    61→            }
    62→
    63→            // Check if chat already exists
    64→            val existing = firestore.collection("chats")
    65→                .whereArrayContains("participants", uid)
    66→                .get().await()
    67→
    68→            for (doc in existing.documents) {
    69→                try {
    70→                    val participants = doc.get("participants") as? List<String> ?: continue
    71→                    if (participants.contains(otherUserId) && !doc.getBoolean("isGroup")) {
    72→                        return Result.success(doc.id)
    73→                    }
    74→                } catch (_: Exception) { continue }
    75→            }
    76→
    77→            // Fetch profiles for chat metadata
    78→            val otherProfile = fetchUserProfile(otherUserId)
    79→            val myProfile = fetchUserProfile(uid)
    80→
    81→            // Create new chat
    82→            val chatId = UUID.randomUUID().toString()
    83→            val chat = Chat(
    84→                id = chatId,
    85→                participants = listOf(uid, otherUserId),
    86→                participantNames = mapOf(uid to myProfile.first, otherUserId to otherProfile.first),
    87→                participantAvatars = mapOf(uid to myProfile.second, otherUserId to otherProfile.second),
    88→                createdAt = System.currentTimeMillis()
    89→            )
    90→            firestore.collection("chats").document(chatId)
    91→                .set(chat.toMap())
    92→                .await()
    93→
    94→            Result.success(chatId)
    95→        } catch (e: Exception) {
    96→            Log.e(TAG, "Failed to get/create chat", e)
    97→            Result.failure(e)
    98→        }
    99→    }

   103→    override suspend fun createGroupChat(groupName: String, participantUids: List<String>): Result<String> {
   104→        return try {
   105→            val uid = auth.currentUser?.uid ?: return Result.failure(Exception("Not authenticated"))
   106→            if (groupName.isBlank()) return Result.failure(Exception("Group name cannot be empty"))
   107→            if (participantUids.isEmpty()) return Result.failure(Exception("Group must have at least one participant"))
   108→
   109→            // ── Contact Gate: Verify all participants are mutual contacts ──
   110→            if (contactRepository != null) {
   111→                for (participantUid in participantUids) {
   112→                    val verification = contactRepository.verifyContactGate(participantUid)
   113→                    if (verification !is ContactVerification.MutualContact) {
   114→                        return Result.failure(Exception("Cannot add user — not a mutual contact"))
   115→                    }
   116→                }
   117→            }
   118→
   119→            val allParticipants = listOf(uid) + participantUids
   120→            val chatId = UUID.randomUUID().toString()
   121→
   122→            // Fetch all participant profiles
   123→            val participantNames = mutableMapOf<String, String>()
   124→            val participantAvatars = mutableMapOf<String, String>()
   125→            for (participantUid in allParticipants) {
   126→                val profile = fetchUserProfile(participantUid)
   127→                participantNames[participantUid] = profile.first
   128→                participantAvatars[participantUid] = profile.second
   129→            }
   130→
   131→            val chat = Chat(
   132→                id = chatId,
   133→                participants = allParticipants,
   134→                participantNames = participantNames,
   135→                participantAvatars = participantAvatars,
   136→                createdAt = System.currentTimeMillis(),
   137→                isGroup = true,
   138→                groupName = groupName
   139→            )
   140→            firestore.collection("chats").document(chatId)
   141→                .set(chat.toMap())
   142→                .await()
   143→
   144→            Result.success(chatId)
   145→        } catch (e: Exception) {
   146→            Log.e(TAG, "Failed to create group chat", e)
   147→            Result.failure(e)
   148→        }
   149→    }

   209→    override suspend fun sendMessage(chatId: String, text: String, replyToId: String): Result<Message> {
   210→        return try {
   211→            val uid = auth.currentUser?.uid ?: return Result.failure(Exception("Not authenticated"))
   212→            val msgId = UUID.randomUUID().toString()
   213→
   214→            // If replying, fetch the replied-to message for metadata
   215→            var replyToText = ""
   216→            var replyToSender = ""
   217→            if (replyToId.isNotBlank()) {
   218→                try {
   219→                    val replyDoc = firestore.collection("chats").document(chatId)
   220→                        .collection("messages").document(replyToId).get().await()
   221→                    replyToText = replyDoc.getString("text") ?: ""
   222→                    replyToSender = replyDoc.getString("senderId") ?: ""
   223→                } catch (_: Exception) {}
   224→            }
   225→
   226→            val message = Message(
   227→                id = msgId,
   228→                chatId = chatId,
   229→                senderId = uid,
   230→                text = text,
   231→                timestamp = System.currentTimeMillis(),
   232→                replyToId = replyToId,
   233→                replyToText = replyToText,
   234→                replyToSender = replyToSender
   235→            )
   236→
   237→            val batch = firestore.batch()
   238→            batch.set(firestore.collection("chats").document(chatId)
   239→                .collection("messages").document(msgId), message.toMap())
   240→            batch.update(firestore.collection("chats").document(chatId), mapOf(
   241→                "lastMessage" to text,
   242→                "lastMessageSenderId" to uid,
   243→                "lastMessageTimestamp" to message.timestamp
   244→            ))
   245→            batch.commit().await()
   246→
   247→            Result.success(message)
   248→        } catch (e: Exception) {
   249→            Log.e(TAG, "Failed to send message", e)
   250→            Result.failure(e)
   251→        }
   252→    }

   451→    override fun startMessagesListening(chatId: String) {
   452→        if (messagesListeners.containsKey(chatId)) return
   453→
   454→        val flow = messagesFlows.getOrPut(chatId) { MutableStateFlow(emptyList()) }
   455→
   456→        val listener = firestore.collection("chats").document(chatId)
   457→            .collection("messages")
   458→            .orderBy("timestamp", Query.Direction.ASCENDING)
   459→            .addSnapshotListener { snapshot, error ->
   460→                if (error != null) {
   461→                    Log.w(TAG, "Messages listener error for $chatId", error)
   462→                    return@addSnapshotListener
   463→                }
   464→                if (snapshot == null) return@addSnapshotListener
   465→
   466→                try {
   467→                    val uid = auth.currentUser?.uid ?: ""
   468→                    val messages = snapshot.documents.mapNotNull { doc ->
   469→                        try { documentToMessage(doc, uid) } catch (_: Exception) { null }
   470→                    }
   471→                    flow.value = messages
   472→                } catch (e: Exception) {
   473→                    Log.e(TAG, "Failed to process messages snapshot", e)
   474→                }
   475→            }
   476→
   477→        messagesListeners[chatId] = listener
   478→    }
   479→
   480→    override fun stopMessagesListening(chatId: String) {
   481→        messagesListeners.remove(chatId)?.remove()
   482→    }

   484→    private fun startChatsListener() {
   485→        val uid = auth.currentUser?.uid ?: return
   486→        chatsListener = firestore.collection("chats")
   487→            .whereArrayContains("participants", uid)
   488→            .addSnapshotListener { snapshot, error ->
   489→                if (error != null) {
   490→                    Log.w(TAG, "Chats listener error", error)
   491→                    return@addSnapshotListener
   492→                }
   493→                if (snapshot == null) return@addSnapshotListener
   494→
   495→                try {
   496→                    val chats = snapshot.documents.mapNotNull { doc ->
   497→                        try { documentToChat(doc) } catch (_: Exception) { null }
   498→                    }.sortedByDescending { it.lastMessageTimestamp }
   499→                    _chats.value = chats
   500→                } catch (e: Exception) {
   501→                    Log.e(TAG, "Failed to process chats snapshot", e)
   502→                }
   503→            }
   504→    }

   506→    private fun startCallsListener() {
   507→        val uid = auth.currentUser?.uid ?: return
   508→
   509→        callsListener = firestore.collection("calls")
   510→            .whereArrayContains("participants", uid)
   511→            .addSnapshotListener { snapshot, error ->
   512→                if (error != null) {
   513→                    Log.w(TAG, "Calls listener error, trying fallback", error)
   514→                    startCallsListenerFallback(uid)
   515→                    return@addSnapshotListener
   516→                }
   517→                if (snapshot == null) return@addSnapshotListener
   518→
   519→                try {
   520→                    val calls = snapshot.documents.mapNotNull { doc ->
   521→                        try { documentToCallRecord(doc) } catch (_: Exception) { null }
   522→                    }.sortedByDescending { it.timestamp }
   523→                    _callHistory.value = calls
   524→                } catch (e: Exception) {
   525→                    Log.e(TAG, "Failed to process calls snapshot", e)
   526→                    _callHistory.value = emptyList()
   527→                }
   528→            }
   529→    }
   530→
   531→    private fun startCallsListenerFallback(uid: String) {
   532→        callsListener?.remove()
   533→        callsListener = firestore.collection("calls")
   534→            .whereEqualTo("callerId", uid)
   535→            .limit(50)
   536→            .addSnapshotListener { snapshot, error ->
   537→                if (error != null) {
   538→                    Log.w(TAG, "Calls fallback listener error", error)
   539→                    startCallsListenerLastResort(uid)
   540→                    return@addSnapshotListener
   541→                }
   542→                if (snapshot == null) return@addSnapshotListener
   543→
   544→                try {
   545→                    val callsAsCaller = snapshot.documents.mapNotNull { doc ->
   546→                        try { documentToCallRecord(doc) } catch (_: Exception) { null }
   547→                    }
   548→
   549→                    firestore.collection("calls")
   550→                        .whereEqualTo("receiverId", uid)
   551→                        .limit(50)
   552→                        .get()
   553→                        .addOnSuccessListener { receiverSnapshot ->
   554→                            val callsAsReceiver = receiverSnapshot.documents.mapNotNull { doc ->
   555→                                try { documentToCallRecord(doc) } catch (_: Exception) { null }
   556→                            }
   557→                            val allCalls = (callsAsCaller + callsAsReceiver)
   558→                                .distinctBy { it.id }
   559→                                .sortedByDescending { it.timestamp }
   560→                            _callHistory.value = allCalls
   561→                        }
   562→                        .addOnFailureListener { e ->
   563→                            Log.w(TAG, "Failed to fetch receiver calls", e)
   564→                            _callHistory.value = callsAsCaller.sortedByDescending { it.timestamp }
   565→                        }
   566→                } catch (e: Exception) {
   567→                    Log.e(TAG, "Failed to process calls fallback snapshot", e)
   568→                    _callHistory.value = emptyList()
   569→                }
   570→            }
   571→    }
   572→
   573→    private fun startCallsListenerLastResort(uid: String) {
   574→        callsListener?.remove()
   575→        scope.launch {
   576→            try {
   577→                val snapshot = firestore.collection("calls")
   578→                    .limit(100)
   579→                    .get()
   580→                    .await()
   581→
   582→                val calls = snapshot.documents.mapNotNull { doc ->
   583→                    try { documentToCallRecord(doc) } catch (_: Exception) { null }
   584→                }.filter { it.callerId == uid || it.receiverId == uid }
   585→                    .sortedByDescending { it.timestamp }
   586→
   587→                _callHistory.value = calls
   588→            } catch (e: Exception) {
   589→                Log.e(TAG, "Last resort calls fetch failed", e)
   590→                _callHistory.value = emptyList()
   591→            }
   592→        }
   593→    }
