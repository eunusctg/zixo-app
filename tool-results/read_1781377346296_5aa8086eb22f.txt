    30→/**
    31→ * Auth Repository Implementation — WebAuthn Passkey Engine
    32→ *
    33→ * Implements the full passkey registration lifecycle:
    34→ * 1. Fetch challenge from Cloudflare Edge Worker at zixo.pages.dev/api/passkey/challenge
    35→ * 2. Defensive JSON validation — enforce rp.id = "zixo.pages.dev" before passing to CredentialManager
    36→ * 3. Invoke Android CredentialManager.createCredential() with CreatePublicKeyCredentialRequest
    37→ * 4. Parse the CreatePublicKeyCredentialResponse and register with backend
    38→ * 5. Save credential reference to Firestore user profile
    39→ *
    40→ * CRITICAL DOMAIN VALIDATION RULES:
    41→ * - rp.id MUST be exactly "zixo.pages.dev" (no protocol, no trailing slash)
    42→ * - rp.name MUST be "Zixo" (hardcoded brand name)
    43→ * - user.id MUST be URL-safe Base64 without padding
    44→ * - challenge MUST be URL-safe Base64 without padding
    45→ * - assetlinks.json MUST be hosted at https://zixo.pages.dev/.well-known/assetlinks.json
    46→ */

   241→    /**
   242→     * Fetch the WebAuthn registration challenge from the Cloudflare Edge Worker.
   243→     *
   244→     * This method makes an HTTP POST to zixo.pages.dev/api/passkey/challenge
   245→     * which returns a properly formatted PublicKeyCredentialCreationOptions JSON.
   246→     *
   247→     * FALLBACK: If the network call fails, generates a local challenge with
   248→     * all the correct domain parameters. This ensures passkey creation works
   249→     * even when the Edge Worker is unreachable.
   250→     */
   251→    override suspend fun getPasskeyRegistrationChallenge(): Result<String> {
   252→        return try {
   253→            val uid = auth.currentUser?.uid ?: return Result.failure(Exception("Not authenticated"))
   254→            val email = auth.currentUser?.email ?: ""
   255→            val displayName = auth.currentUser?.displayName ?: "Zixo User"
   256→
   257→            // ── Attempt to fetch from Cloudflare Edge Worker ──
   258→            try {
   259→                val challengeJson = fetchChallengeFromEdgeWorker(uid, email, displayName)
   260→                if (challengeJson != null) {
   261→                    // ── DEFENSIVE VALIDATION: Enforce rp.id before returning ──
   262→                    val validatedJson = enforceRpIdCompliance(challengeJson)
   263→                    Log.d(TAG, "Challenge fetched from Edge Worker, rp.id validated")
   264→                    return Result.success(validatedJson)
   265→                }
   266→            } catch (e: Exception) {
   267→                Log.w(TAG, "Edge Worker challenge fetch failed, using local fallback", e)
   268→            }
   269→
   270→            // ── LOCAL FALLBACK: Generate challenge with correct parameters ──
   271→            val localChallenge = generateLocalChallenge(uid, email, displayName)
   272→            Result.success(localChallenge)
   273→        } catch (e: Exception) {
   274→            Log.e(TAG, "Failed to get passkey challenge", e)
   275→            Result.failure(e)
   276→        }
   277→    }

   279→    /**
   280→     * Fetch challenge JSON from the Cloudflare Edge Worker endpoint.
   281→     * Runs on IO thread.
   282→     */
   283→    private suspend fun fetchChallengeFromEdgeWorker(
   284→        uid: String,
   285→        email: String,
   286→        displayName: String
   287→    ): String? = withContext(Dispatchers.IO) {
   288→        try {
   289→            val url = URL(PASSKEY_CHALLENGE_URL)
   290→            val connection = url.openConnection() as HttpURLConnection
   291→            connection.requestMethod = "POST"
   292→            connection.setRequestProperty("Content-Type", "application/json")
   293→            connection.setRequestProperty("Accept", "application/json")
   294→            connection.connectTimeout = 10000
   295→            connection.readTimeout = 15000
   296→            connection.doOutput = true
   297→
   298→            val requestBody = JSONObject().apply {
   299→                put("uid", uid)
   300→                put("email", email)
   301→                put("displayName", displayName)
   302→            }
   303→
   304→            connection.outputStream.use { os ->
   305→                val input = requestBody.toString().toByteArray(Charsets.UTF_8)
   306→                os.write(input, 0, input.size)
   307→            }
   308→
   309→            val responseCode = connection.responseCode
   310→            if (responseCode == HttpURLConnection.HTTP_OK) {
   311→                val response = connection.inputStream.bufferedReader().use { it.readText() }
   312→                connection.disconnect()
   313→                response
   314→            } else {
   315→                Log.w(TAG, "Edge Worker returned HTTP $responseCode")
   316→                connection.disconnect()
   317→                null
   318→            }
   319→        } catch (e: Exception) {
   320→            Log.w(TAG, "Edge Worker fetch exception: ${e.message}")
   321→            null
   322→        }
   323→    }

   325→    /**
   326→     * DEFENSIVE VALIDATION: Enforce that the rp.id in the challenge JSON
   327→     * is exactly "zixo.pages.dev". This prevents the "rp id cannot be validated"
   328→     * error that occurs when the backend returns an incorrect domain.
   329→     *
   330→     * This method:
   331→     * 1. Parses the JSON
   332→     * 2. Checks rp.id against the hardcoded RP_ID constant
   333→     * 3. Overwrites rp.id with the correct value if it doesn't match
   334→     * 4. Ensures rp.name is "Zixo"
   335→     * 5. Returns the corrected JSON string
   336→     */
   337→    private fun enforceRpIdCompliance(challengeJson: String): String {
   338→        return try {
   339→            val json = JSONObject(challengeJson)
   340→
   341→            // ── Validate and correct rp object ──
   342→            val rp = json.optJSONObject("rp") ?: JSONObject()
   343→            val currentRpId = rp.optString("id", "")
   344→            val currentRpName = rp.optString("name", "")
   345→
   346→            if (currentRpId != RP_ID) {
   347→                Log.w(TAG, "RP ID mismatch detected! Server returned: '$currentRpId', enforcing: '$RP_ID'")
   348→                rp.put("id", RP_ID)
   349→            }
   350→
   351→            if (currentRpName != RP_NAME) {
   352→                Log.w(TAG, "RP name mismatch detected! Server returned: '$currentRpName', enforcing: '$RP_NAME'")
   353→                rp.put("name", RP_NAME)
   354→            }
   355→
   356→            json.put("rp", rp)
   357→
   358→            // ── Ensure user.id is Base64Url encoded ──
   359→            val user = json.optJSONObject("user")
   360→            if (user != null) {
   361→                val userId = user.optString("id", "")
   362→                // Check if the user.id looks like a raw UUID/string instead of Base64Url
   363→                if (userId.isNotEmpty() && !isValidBase64Url(userId)) {
   364→                    Log.w(TAG, "user.id is not Base64Url encoded, encoding it now")
   365→                    user.put("id", encodeToBase64Url(userId))
   366→                }
   367→            }
   368→
   369→            // ── Ensure challenge is Base64Url encoded ──
   370→            val challenge = json.optString("challenge", "")
   371→            if (challenge.isNotEmpty() && !isValidBase64Url(challenge)) {
   372→                Log.w(TAG, "Challenge is not Base64Url encoded, encoding it now")
   373→                json.put("challenge", encodeToBase64Url(challenge))
   374→            }
   375→
   376→            json.toString()
   377→        } catch (e: Exception) {
   378→            Log.e(TAG, "Failed to validate/enforce RP ID compliance", e)
   379→            // If JSON parsing fails, do a regex-based fix as last resort
   380→            challengeJson
   381→                .replace("\"id\"\\s*:\\s*\"[^\"]*\"".toRegex(RegexOption.IGNORE_CASE), "\"id\": \"$RP_ID\"")
   382→        }
   383→    }

   385→    /**
   386→     * Generate a local WebAuthn challenge with all correct domain parameters.
   387→     * Used as fallback when the Cloudflare Edge Worker is unreachable.
   388→     *
   389→     * CRITICAL: Every field follows the exact WebAuthn spec for Android:
   390→     * - challenge: 32 bytes of cryptographic randomness, Base64Url encoded, no padding
   391→     * - rp.id: exactly "zixo.pages.dev"
   392→     * - rp.name: "Zixo"
   393→     * - user.id: Base64Url encoded UID, no padding
   394→     */
   395→    private fun generateLocalChallenge(uid: String, email: String, displayName: String): String {
   396→        val challengeBytes = ByteArray(32)
   397→        java.security.SecureRandom().nextBytes(challengeBytes)
   398→        val challenge = encodeToBase64Url(challengeBytes)
   399→
   400→        val userIdBase64 = encodeToBase64Url(uid.toByteArray(Charsets.UTF_8))
   401→
   402→        val json = JSONObject().apply {
   403→            put("challenge", challenge)
   404→            put("rp", JSONObject().apply {
   405→                put("name", RP_NAME)
   406→                put("id", RP_ID)
   407→            })
   408→            put("user", JSONObject().apply {
   409→                put("id", userIdBase64)
   410→                put("name", email.ifBlank { "$uid@zixo.pages.dev" })
   411→                put("displayName", displayName.ifBlank { "Zixo User" })
   412→            })
   413→            put("pubKeyCredParams", org.json.JSONArray().apply {
   414→                put(JSONObject().put("type", "public-key").put("alg", -7))
   415→                put(JSONObject().put("type", "public-key").put("alg", -257))
   416→            })
   417→            put("timeout", 180000)
   418→            put("excludeCredentials", org.json.JSONArray())
   419→            put("authenticatorSelection", JSONObject().apply {
   420→                put("authenticatorAttachment", "platform")
   421→                put("requireResidentKey", true)
   422→                put("residentKey", "required")
   423→                put("userVerification", "required")
   424→            })
   425→            put("attestation", "none")
   426→        }
   427→
   428→        Log.d(TAG, "Local challenge generated with rp.id=$RP_ID, user.id=${userIdBase64.take(20)}...")
   429→        return json.toString()
   430→    }

   537→    /**
   538→     * Encode a byte array to URL-safe Base64 without padding.
   539→     * This is the format required by WebAuthn for challenge and user.id fields.
   540→     *
   541→     * The Android CredentialManager expects Base64Url encoding (RFC 4648 §5):
   542→     * - Replace '+' with '-'
   543→     * - Replace '/' with '_'
   544→     * - Remove trailing '=' padding
   545→     */
   546→    private fun encodeToBase64Url(data: ByteArray): String {
   547→        return Base64.encodeToString(data, Base64.URL_SAFE or Base64.NO_PADDING or Base64.NO_WRAP)
   548→    }
   549→
   550→    private fun encodeToBase64Url(str: String): String {
   551→        return encodeToBase64Url(str.toByteArray(Charsets.UTF_8))
   552→    }

   554→    /**
   555→     * Check if a string is valid Base64Url encoding.
   556→     * Valid Base64Url uses only: A-Z, a-z, 0-9, -, _ (no +, /, or =)
   557→     */
   558→    private fun isValidBase64Url(str: String): Boolean {
   559→        return str.isNotEmpty() && str.all { it.isLetterOrDigit() || it == '-' || it == '_' }
   560→    }

   566→    private suspend fun fetchUserProfile(uid: String): UserProfile {
   567→        val doc = firestore.collection("users").document(uid).get().await()
   568→        if (!doc.exists()) throw Exception("User profile not found")

   572→    private suspend fun fetchOrCreateUserProfile(uid: String, displayName: String, photoUrl: String): UserProfile {
   573→        return try {
   574→            val doc = firestore.collection("users").document(uid).get().await()
   575→            if (doc.exists()) {
   576→                documentToProfile(doc)
   577→            } else {
   578→                createNewUserProfile(uid, displayName, photoUrl)
   579→            }
   580→        } catch (e: Exception) {
   581→            Log.w(TAG, "Fetch/create failed, creating new profile", e)
   582→            createNewUserProfile(uid, displayName, photoUrl)
   583→        }
   584→    }

   586→    private suspend fun createNewUserProfile(uid: String, displayName: String, photoUrl: String): UserProfile {
   587→        val zixoNumber = generateZixoNumber()
   588→        val username = generateUsername(displayName)
   589→
   590→        val profile = UserProfile(
   591→            uid = uid,
   592→            displayName = displayName.ifBlank { "Zixo User" },
   593→            username = username,
   594→            zixoNumber = zixoNumber,
   595→            phoneNumber = "",
   596→            bio = "",
   597→            avatarUrl = photoUrl,
   598→            createdAt = System.currentTimeMillis(),
   599→            isOnline = true,
   600→            lastSeen = System.currentTimeMillis()
   601→        )
   602→
   603→        try {
   604→            firestore.collection("users").document(uid)
   605→                .set(profile.toMap(), SetOptions.merge())
   606→                .await()
   607→        } catch (e: Exception) {
   608→            Log.e(TAG, "Failed to save new user profile to Firestore", e)
   609→        }
   610→
   611→        return profile
   612→    }

   614→    /** Generate 8-digit Zixo Number formatted as "XXXX XXXX" */
   615→    private suspend fun generateZixoNumber(): String {
   616→        var attempts = 0
   617→        while (attempts < 20) {
   618→            val raw = Random.nextInt(10000000, 99999999).toString()
   619→            val formatted = "${raw.substring(0, 4)} ${raw.substring(4)}"
   620→            try {
   621→                val existing = firestore.collection("users")
   622→                    .whereEqualTo("zixoNumber", formatted)
   623→                    .limit(1).get().await()
   624→                if (existing.isEmpty) return formatted
   625→            } catch (e: Exception) {
   626→                return formatted
   627→            }
   628→            attempts++
   629→        }
   630→        val raw = System.currentTimeMillis().toString().takeLast(8)
   631→        return "${raw.substring(0, 4)} ${raw.substring(4)}"
   632→    }

   634→    private fun generateUsername(displayName: String): String {
   635→        val base = displayName.lowercase().replace(Regex("[^a-z0-9]"), "").take(8)
   636→        val suffix = Random.nextInt(100, 999)
   637→        return if (base.isNotBlank()) "zixo_${base}$suffix" else "zixo_user$suffix"
   638→    }

   655→    private suspend fun loadCachedProfile() {

   657→            dataStore.getCachedProfile().collect { cached ->

   659→                    _authState.value = AuthState.Unauthenticated
   660→                } else {

   662→                        uid = cached["uid"] ?: "",

   664→                        username = cached["username"] ?: "",
   665→                        zixoNumber = cached["zixoNumber"] ?: "",
   666→                        phoneNumber = cached["phoneNumber"] ?: "",

   668→                        avatarUrl = cached["avatarUrl"] ?: ""
