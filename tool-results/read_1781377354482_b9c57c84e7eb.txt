   259‚Üí   259‚Üí                val challengeJson = fetchChallengeFromEdgeWorker(uid, email, displayName)

   268‚Üí   268‚Üí            }

   614‚Üí   614‚Üí    /** Generate 8-digit Zixo Number formatted as "XXXX XXXX" */
   615‚Üí   615‚Üí    private suspend fun generateZixoNumber(): String {
   616‚Üí   616‚Üí        var attempts = 0
   617‚Üí   617‚Üí        while (attempts < 20) {
   618‚Üí   618‚Üí            val raw = Random.nextInt(10000000, 99999999).toString()
   619‚Üí   619‚Üí            val formatted = "${raw.substring(0, 4)} ${raw.substring(4)}"
   620‚Üí   620‚Üí            try {
   621‚Üí   621‚Üí                val existing = firestore.collection("users")
   622‚Üí   622‚Üí                    .whereEqualTo("zixoNumber", formatted)
   623‚Üí   623‚Üí                    .limit(1).get().await()
   624‚Üí   624‚Üí                if (existing.isEmpty) return formatted
   625‚Üí   625‚Üí            } catch (e: Exception) {
   626‚Üí   626‚Üí                return formatted
   627‚Üí   627‚Üí            }
   628‚Üí   628‚Üí            attempts++
   629‚Üí   629‚Üí        }
   630‚Üí   630‚Üí        val raw = System.currentTimeMillis().toString().takeLast(8)
   631‚Üí   631‚Üí        return "${raw.substring(0, 4)} ${raw.substring(4)}"
   632‚Üí   632‚Üí    }
