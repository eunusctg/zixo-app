    50→@OptIn(ExperimentalMaterial3Api::class)
    51→@Composable
    52→fun HomeScreen(
    53→    navController: NavHostController,
    54→    viewModel: HomeViewModel = hiltViewModel()
    55→) {
    56→    val selectedTab by viewModel.selectedTab.collectAsState()
    57→    val currentUser by viewModel.currentUser.collectAsState()
    58→    val searchQuery by viewModel.searchQuery.collectAsState()
    59→    val isSearching by viewModel.isSearching.collectAsState()
    60→
    61→    val pagerState = rememberPagerState(
    62→        initialPage = 0,
    63→        pageCount = { HomeTab.entries.size }
    64→    )
    65→    val scope = rememberCoroutineScope()
    66→
    67→    // Sync pager ↔ tabs
    68→    LaunchedEffect(pagerState.currentPage) {
    69→        viewModel.selectTab(HomeTab.entries[pagerState.currentPage])
    70→    }
    71→    LaunchedEffect(selectedTab) {
    72→        val idx = HomeTab.entries.indexOf(selectedTab)
    73→        if (pagerState.currentPage != idx) {
    74→            pagerState.animateScrollToPage(idx)
    75→        }
    76→    }
    77→
    78→    // FAB expanded state
    79→    var fabExpanded by remember { mutableStateOf(false) }
    80→
    81→    Scaffold(
    82→        containerColor = ZexoBackground,
    83→        topBar = {
    84→            HomeTopBar(
    85→                currentUser = currentUser,
    86→                searchQuery = searchQuery,
    87→                onSearchQueryChange = { viewModel.updateSearchQuery(it) },
    88→                onClearSearch = { viewModel.clearSearch() },
    89→                isSearching = isSearching,
    90→                onAvatarClick = {
    91→                    currentUser?.uid?.let { uid ->
    92→                        navController.navigate(Screen.Profile.createRoute(uid))
    93→                    }
    94→                },
    95→                onAdminClick = {
    96→                    navController.navigate(Screen.AdminLanding.route)
    97→                },
    98→                isAdmin = viewModel.isAdmin
    99→            )
   100→        },
   101→        floatingActionButton = {
   102→            AnimatedFab(
   103→                selectedTab = selectedTab,
   104→                expanded = fabExpanded,
   105→                onExpandToggle = { fabExpanded = !fabExpanded },
   106→                onDismiss = { fabExpanded = false },
   107→                navController = navController
   108→            )
   109→        }
   110→    ) { paddingValues ->
   111→        Column(
   112→            modifier = Modifier
   113→                .fillMaxSize()
   114→                .padding(paddingValues)
   115→                .background(ZexoBackground)
   116→        ) {
   117→            // Tab row
   118→            ZexoTabRow(
   119→                selectedTab = selectedTab,
   120→                onTabSelected = { tab ->
   121→                    viewModel.selectTab(tab)
   122→                    scope.launch {
   123→                        val idx = HomeTab.entries.indexOf(tab)
   124→                        pagerState.animateScrollToPage(idx)
   125→                    }
   126→                }
   127→            )
   128→
   129→            // Pager
   130→            HorizontalPager(
   131→                state = pagerState,
   132→                modifier = Modifier.fillMaxSize(),
   133→                beyondViewportPageCount = 3
   134→            ) { page ->
   135→                when (HomeTab.entries[page]) {
   136→                    HomeTab.CHATS -> ChatsTabContent(viewModel, navController)
   137→                    HomeTab.STATUS -> StatusTabContent(viewModel, navController)
   138→                    HomeTab.CALLS -> CallsTabContent(viewModel, navController)
   139→                    HomeTab.SETTINGS -> SettingsTabContent(viewModel, navController)
   140→                }
   141→            }
   142→        }
   143→    }
   144→
   145→    // Dismiss FAB menu on tab change
   146→    LaunchedEffect(selectedTab) {
   147→        fabExpanded = false
   148→    }
   149→}

   280→                                modifier = Modifier.size(18.dp),

   283→                            )

   285→                    },
   286→                    singleLine = true,
   287→                    shape = RoundedCornerShape(12.dp),

   289→                        focusedBorderColor = ZexoPrimary,
   290→                        unfocusedBorderColor = ZexoSurfaceLight,
   291→                        focusedContainerColor = ZexoBackground,

   293→                        focusedTextColor = ZexoTextPrimary,
   294→                        unfocusedTextColor = ZexoTextPrimary,

   297→                )
   298→            }
   299→        }
   300→    }

   334→@Composable
   335→private fun ZexoTabItem(
   336→    tab: HomeTab,
   337→    selected: Boolean,
   338→    onClick: () -> Unit
   339→) {
   340→    val icon: ImageVector = when (tab) {
   341→        HomeTab.CHATS -> Icons.Default.Chat
   342→        HomeTab.STATUS -> Icons.Default.CameraAlt
   343→        HomeTab.CALLS -> Icons.Default.Call
   344→        HomeTab.SETTINGS -> Icons.Default.Settings
   345→    }
   346→    val selectedIcon: ImageVector = when (tab) {
   347→        HomeTab.CHATS -> Icons.Filled.Chat
   348→        HomeTab.STATUS -> Icons.Filled.CameraAlt
   349→        HomeTab.CALLS -> Icons.Filled.Call
   350→        HomeTab.SETTINGS -> Icons.Filled.Settings
   351→    }
   352→
   353→    val animatedColor by animateColorAsState(
   354→        targetValue = if (selected) ZexoPrimary else ZexoTextSecondary,
   355→        animationSpec = tween(250), label = "tabColor"
   356→    )
   357→
   358→    Column(
   359→        modifier = Modifier
   360→            .clip(RoundedCornerShape(10.dp))
   361→            .clickable(onClick = onClick)
   362→            .background(if (selected) ZexoPrimary.copy(alpha = 0.12f) else Color.Transparent)
   363→            .padding(horizontal = 20.dp, vertical = 8.dp),
   364→        horizontalAlignment = Alignment.CenterHorizontally
   365→    ) {
   366→        Icon(
   367→            imageVector = if (selected) selectedIcon else icon,
   368→            contentDescription = tab.label,
   369→            tint = animatedColor,
   370→            modifier = Modifier.size(22.dp)
   371→        )
   372→        Spacer(modifier = Modifier.height(2.dp))
   373→        Text(
   374→            text = tab.label,
   375→            fontSize = 11.sp,
   376→            fontWeight = if (selected) FontWeight.SemiBold else FontWeight.Normal,
   377→            color = animatedColor
   378→        )
   379→        // Indicator dot
   380→        AnimatedVisibility(visible = selected) {
   381→            Box(
   382→                modifier = Modifier
   383→                    .padding(top = 2.dp)
   384→                    .width(16.dp)
   385→                    .height(3.dp)
   386→                    .clip(RoundedCornerShape(2.dp))
   387→                    .background(ZexoPrimary)
   388→            )
   389→        }
   390→    }
   391→}

   393→// ═══════════════════════════════════════════════════════════════════════
   394→//  Animated FAB with dropdown menus
   395→// ═══════════════════════════════════════════════════════════════════════
   396→
   397→@Composable
   398→private fun AnimatedFab(
   399→    selectedTab: HomeTab,
   400→    expanded: Boolean,
   401→    onExpandToggle: () -> Unit,
   402→    onDismiss: () -> Unit,
   403→    navController: NavHostController
   404→) {
   405→    // Only show FAB on Chats and Calls tabs
   406→    if (selectedTab != HomeTab.CHATS && selectedTab != HomeTab.CALLS) return
   407→
   408→    val rotation by animateFloatAsState(
   409→        targetValue = if (expanded) 45f else 0f,
   410→        animationSpec = tween(300),
   411→        label = "fabRotation"
   412→    )
   413→
   414→    val fabIcon: ImageVector = when (selectedTab) {
   415→        HomeTab.CHATS -> Icons.Default.Chat
   416→        HomeTab.CALLS -> Icons.Default.Call
   417→        else -> Icons.Default.Add
   418→    }
   419→
   420→    val fabColor = when (selectedTab) {
   421→        HomeTab.CHATS -> ZexoPrimary
   422→        HomeTab.CALLS -> ZexoSecondary
   423→        else -> ZexoPrimary
   424→    }
   425→
   426→    Box(
   427→        modifier = Modifier.padding(end = 16.dp, bottom = 16.dp)
   428→    ) {
   429→        // Dimming overlay
   430→        AnimatedVisibility(
   431→            visible = expanded,
   432→            enter = fadeIn(tween(200)),
   433→            exit = fadeOut(tween(200))
   434→        ) {
   435→            // Invisible click-away layer – handled by onDismiss calls elsewhere
   436→        }
   437→
   438→        // Dropdown menu items (shown above FAB)
   439→        AnimatedVisibility(
   440→            visible = expanded,
   441→            enter = fadeIn(tween(200)) + slideInVertically(initialOffsetY = { it / 2 }),
   442→            exit = fadeOut(tween(150)) + slideOutVertically(targetOffsetY = { it / 2 })
   443→        ) {
   444→            Column(
   445→                horizontalAlignment = Alignment.End,
   446→                verticalArrangement = Arrangement.spacedBy(10.dp),
   447→                modifier = Modifier.padding(bottom = 12.dp)
   448→            ) {
   449→                when (selectedTab) {
   450→                    HomeTab.CHATS -> ChatFabMenu(navController, onDismiss)
   451→                    HomeTab.CALLS -> CallsFabMenu(navController, onDismiss)
   452→                    else -> {}
   453→                }
   454→            }
   455→        }
   456→
   457→        // Main FAB button
   458→        FloatingActionButton(
   459→            onClick = onExpandToggle,
   460→            containerColor = fabColor,
   461→            shape = CircleShape,
   462→            elevation = FloatingActionButtonDefaults.elevation(
   463→                defaultElevation = 6.dp,
   464→                pressedElevation = 10.dp
   465→            )
   466→        ) {
   467→            Icon(
   468→                imageVector = Icons.Default.Add,
   469→                contentDescription = "Action",
   470→                tint = Color.White,
   471→                modifier = Modifier.rotate(rotation)
   472→            )
   473→        }
   474→    }
   475→}

   586→
   587→@Composable
   588→private fun ChatsTabContent(
   589→    viewModel: HomeViewModel,
   590→    navController: NavHostController
   591→) {
   592→    val chats by viewModel.filteredChats.collectAsState()
   593→    val searchResults by viewModel.searchResults.collectAsState()
   594→    val searchQuery by viewModel.searchQuery.collectAsState()
   595→    val isLoading by viewModel.isLoading.collectAsState()
   596→
   597→    if (isLoading) {
   598→        Box(
   599→            modifier = Modifier.fillMaxSize(),
   600→            contentAlignment = Alignment.Center
   601→        ) {
   602→            CircularProgressIndicator(color = ZexoPrimary)
   603→        }
   604→        return
   605→    }
   606→
   607→    // Show search results if searching
   608→    if (searchQuery.isNotBlank()) {
   609→        if (searchResults.isEmpty()) {
   610→            Box(
   611→                modifier = Modifier.fillMaxSize(),
   612→                contentAlignment = Alignment.Center
   613→            ) {
   614→                Column(horizontalAlignment = Alignment.CenterHorizontally) {
   615→                    Icon(
   616→                        Icons.Default.Search,
   617→                        contentDescription = null,
   618→                        modifier = Modifier.size(48.dp),
   619→                        tint = ZexoTextSecondary
   620→                    )
   621→                    Spacer(modifier = Modifier.height(8.dp))
   622→                    Text(
   623→                        "No results for \"$searchQuery\"",
   624→                        color = ZexoTextSecondary,
   625→                        fontSize = 14.sp
   626→                    )
   627→                }
   628→            }
   629→        } else {
   630→            LazyColumn(
   631→                modifier = Modifier.fillMaxSize(),
   632→                contentPadding = PaddingValues(vertical = 8.dp)
   633→            ) {
   634→                items(searchResults, key = { it.uid }) { user ->
   635→                    SearchUserItem(
   636→                        user = user,
   637→                        isOnline = viewModel.isUserOnline(user.uid),
   638→                        onClick = {
   639→                            navController.navigate(Screen.Profile.createRoute(user.uid))
   640→                        }
   641→                    )
   642→                }
   643→            }
   644→        }

  1525→private fun formatTimestamp(ts: Long): String {
  1526→    val now = System.currentTimeMillis()
  1527→    val diff = now - ts
  1528→    return when {
  1529→        diff < 60_000 -> "now"
  1530→        diff < 3_600_000 -> "${diff / 60_000}m"
  1531→        diff < 86_400_000 -> "${diff / 3_600_000}h"
  1532→        diff < 604_800_000 -> "${diff / 86_400_000}d"
  1533→        else -> {
  1534→            val date = java.text.SimpleDateFormat("MMM dd", java.util.Locale.getDefault())
  1535→            date.format(java.util.Date(ts))
  1536→        }
  1537→    }
  1538→}
  1539→
  1540→private fun formatCallInfo(record: CallRecord): String {
  1541→    val time = record.timestamp?.let { formatTimestamp(it) } ?: ""
  1542→    val duration = if (record.duration > 0) {
  1543→        val mins = record.duration / 60
  1544→        val secs = record.duration % 60
  1545→        "($mins:${secs.toString().padStart(2, '0')})"
  1546→    } else ""
  1547→    val direction = when (record.direction) {
  1548→        "outgoing" -> "Outgoing"
  1549→        "incoming" -> "Incoming"
  1550→        "missed" -> "Missed"
  1551→        else -> record.direction.replaceFirstChar { it.uppercase() }
  1552→    }
  1553→    return "$direction $duration $time".trim()
  1554→}
